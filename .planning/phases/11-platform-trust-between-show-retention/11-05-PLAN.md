**trending_repository.dart:**
Create following DioClient pattern (see rsvp_repository.dart or onboarding_repository.dart for reference).

```dart
class TrendingRepository {
  final DioClient _client;
  TrendingRepository(this._client);

  Future<List<TrendingEvent>> getTrendingNearby({
    required double lat,
    required double lon,
    int radius = 80,
    int days = 30,
    int limit = 20,
  }) async {
    final response = await _client.get('/api/trending', queryParameters: {
      'lat': lat, 'lon': lon, 'radius': radius, 'days': days, 'limit': limit,
    });
    final List<dynamic> data = response.data['data'];
    return data.map((e) => TrendingEvent.fromJson(e)).toList();
  }
}
```

Create `SearchResults` model:
```dart
class SearchResults {
  final List<Band> bands;
  final List<Venue> venues;
  final List<Event> events;

  bool get isEmpty => bands.isEmpty && venues.isEmpty && events.isEmpty;

  factory SearchResults.empty() => SearchResults(bands: [], venues: [], events: []);
  factory SearchResults.fromJson(Map<String, dynamic> json) => SearchResults(
    bands: (json['bands'] as List?)?.map((e) => Band.fromJson(e)).toList() ?? [],
```
