**claim_repository.dart:**
Create following DioClient pattern.

```dart
class ClaimRepository {
  final DioClient _client;
  ClaimRepository(this._client);

  Future<VerificationClaim> submitClaim({
    required String entityType,
    required String entityId,
    String? evidenceText,
    String? evidenceUrl,
  }) async {
    final response = await _client.post('/api/claims', data: {
      'entityType': entityType, 'entityId': entityId,
      if (evidenceText != null) 'evidenceText': evidenceText,
      if (evidenceUrl != null) 'evidenceUrl': evidenceUrl,
    });
    return VerificationClaim.fromJson(response.data['data']);
  }

  Future<List<VerificationClaim>> getMyClaims() async {
    final response = await _client.get('/api/claims/me');
    return (response.data['data'] as List).map((e) => VerificationClaim.fromJson(e)).toList();
  }

  Future<Map<String, dynamic>> getEntityStats(String entityType, String entityId) async {
    final response = await _client.get('/api/claims/stats/$entityType/$entityId');
    return response.data['data'];
  }
}
```
