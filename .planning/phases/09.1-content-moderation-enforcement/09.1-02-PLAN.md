```typescript
// The helper method:
private async invalidateFeedCachesForCheckin(userId: string, eventId: string | null): Promise<void> {
  // Gets all followers of the user
  const followers = await this.db.query(
    'SELECT follower_id FROM user_followers WHERE following_id = $1',
    [userId]
  );
  for (const f of followers.rows) {
    await this.feedService.invalidateUserFeedCache(f.follower_id);
  }
  if (eventId) {
    await this.feedService.invalidateEventFeedCache(eventId);
  }
}
```
