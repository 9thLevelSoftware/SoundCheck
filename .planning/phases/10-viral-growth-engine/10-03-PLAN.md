    **OnboardingRepository (new file):**
    Follow existing repository pattern (uses DioClient):
    ```dart
    class OnboardingRepository {
      final Dio _dio;
      OnboardingRepository(this._dio);

      Future<void> saveGenrePreferences(List<String> genres) async {
        await _dio.post('/api/onboarding/genres', data: {'genres': genres});
      }

      Future<List<String>> getGenrePreferences() async {
        final response = await _dio.get('/api/onboarding/genres');
        return List<String>.from(response.data['data']['genres'] ?? []);
      }

      Future<void> completeOnboarding() async {
        await _dio.post('/api/onboarding/complete');
      }

      Future<bool> isOnboardingComplete() async {
        final response = await _dio.get('/api/onboarding/status');
        return response.data['data']['completed'] ?? false;
      }
    }
    ```
