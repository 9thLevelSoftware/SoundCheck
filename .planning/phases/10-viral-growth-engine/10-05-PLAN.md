    ```dart
    class RsvpRepository {
      final Dio _dio;
      RsvpRepository(this._dio);

      /// Toggle RSVP for an event (creates or deletes)
      Future<bool> toggleRsvp(String eventId) async {
        final response = await _dio.post('/api/rsvp/$eventId');
        return response.data['data']['isGoing'] as bool;
      }

      /// Get friends who RSVP'd to an event
      Future<FriendsGoingData> getFriendsGoing(String eventId) async {
        final response = await _dio.get('/api/rsvp/$eventId/friends');
        final data = response.data['data'];
        return FriendsGoingData(
          count: data['count'] as int,
          friends: (data['friends'] as List).map((f) => FriendAvatar(
            id: f['id'] as String,
            username: f['username'] as String,
            profileImageUrl: f['profileImageUrl'] as String?,
          )).toList(),
        );
      }

      /// Get all event IDs user has RSVP'd to (for batch status checking)
      Future<Set<String>> getUserRsvps() async {
        final response = await _dio.get('/api/rsvp/me');
        final eventIds = response.data['data']['eventIds'] as List;
        return eventIds.map((e) => e as String).toSet();
      }
    }
    ```

    ```dart
    class RsvpButton extends ConsumerWidget {
      final String eventId;
      const RsvpButton({super.key, required this.eventId});

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        final userRsvps = ref.watch(userRsvpsProvider);
        final isGoing = userRsvps.valueOrNull?.contains(eventId) ?? false;

        return AnimatedContainer(
          duration: const Duration(milliseconds: 200),
          child: ElevatedButton.icon(
            onPressed: () => _toggleRsvp(ref),
            icon: Icon(isGoing ? Icons.check_circle : Icons.add_circle_outline),
            label: Text(isGoing ? "I'm Going!" : "I'm Going"),
            style: ElevatedButton.styleFrom(
              backgroundColor: isGoing ? AppTheme.primaryColor : Colors.transparent,
              foregroundColor: isGoing ? Colors.white : AppTheme.primaryColor,
              side: BorderSide(color: AppTheme.primaryColor),
            ),
          ),
        );
      }

      Future<void> _toggleRsvp(WidgetRef ref) async {
        try {
          await ref.read(rsvpRepositoryProvider).toggleRsvp(eventId);
          // Invalidate to refresh RSVP state
          ref.invalidate(userRsvpsProvider);
          ref.invalidate(friendsGoingProvider(eventId));
        } catch (e) {
          // Show error snackbar
        }
      }
    }
    ```

    ```dart
    class FriendsGoingWidget extends ConsumerWidget {
      final String eventId;
      const FriendsGoingWidget({super.key, required this.eventId});

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        final friendsGoing = ref.watch(friendsGoingProvider(eventId));

        return friendsGoing.when(
          data: (data) {
            if (data.count == 0) return const SizedBox.shrink();
            return Row(
              children: [
                // Overlapping avatar stack (max 5 shown)
                SizedBox(
                  width: (data.friends.length.clamp(0, 5) * 24) + 8,
                  height: 32,
                  child: Stack(
                    children: data.friends.take(5).toList().asMap().entries.map((entry) {
                      return Positioned(
                        left: entry.key * 20.0,
                        child: CircleAvatar(
                          radius: 16,
                          backgroundImage: entry.value.profileImageUrl != null
                            ? CachedNetworkImageProvider(entry.value.profileImageUrl!)
                            : null,
                          child: entry.value.profileImageUrl == null
                            ? Text(entry.value.username[0].toUpperCase())
                            : null,
                        ),
                      );
                    }).toList(),
                  ),
                ),
                const SizedBox(width: 8),
                // Count text
                Text(
                  data.count == 1
                    ? '1 friend going'
                    : '${data.count} friends going',
                  style: Theme.of(context).textTheme.bodySmall?.copyWith(
                    color: AppTheme.primaryColor,
                  ),
                ),
              ],
            );
          },
          loading: () => const SizedBox.shrink(), // Don't show loading for this
          error: (_, __) => const SizedBox.shrink(), // Silently hide on error
        );
      }
    }
    ```
