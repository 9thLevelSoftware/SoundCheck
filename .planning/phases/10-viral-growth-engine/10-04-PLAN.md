    ```dart
    class ShareRepository {
      final Dio _dio;
      ShareRepository(this._dio);

      /// Request server to generate check-in share card images
      Future<ShareCardUrls> generateCheckinCard(String checkinId) async {
        final response = await _dio.post('/api/share/checkin/$checkinId');
        final data = response.data['data'];
        return ShareCardUrls(
          ogUrl: data['ogUrl'] as String,
          storiesUrl: data['storiesUrl'] as String,
        );
      }

      /// Request server to generate badge share card images
      Future<ShareCardUrls> generateBadgeCard(String badgeAwardId) async {
        final response = await _dio.post('/api/share/badge/$badgeAwardId');
        final data = response.data['data'];
        return ShareCardUrls(
          ogUrl: data['ogUrl'] as String,
          storiesUrl: data['storiesUrl'] as String,
        );
      }
    }

    class ShareCardUrls {
      final String ogUrl;
      final String storiesUrl;
      const ShareCardUrls({required this.ogUrl, required this.storiesUrl});
    }
    ```

    ```dart
    class SocialShareService {
      /// Share to Instagram Stories (requires 1080x1920 image)
      static Future<bool> shareToInstagramStories(String imageUrl) async {
        try {
          // Download image to temp file first
          final tempFile = await _downloadToTemp(imageUrl, 'stories');
          // Use social_share_kit if available, else fall back to share_plus
          // await SocialShareKit.shareToInstagramStories(imagePath: tempFile.path);
          // Fallback: use share_plus with the image
          await SharePlus.instance.share(ShareParams(files: [XFile(tempFile.path)]));
          return true;
        } catch (e) {
          return false;
        }
      }

      /// Share to TikTok
      static Future<bool> shareToTikTok(String imageUrl) async {
        // Similar pattern -- download, share via social_share_kit or fallback
      }

      /// Share via OS share sheet (X/Twitter, generic)
      static Future<bool> shareGeneric({
        required String text,
        required String imageUrl,
        required String shareUrl,
      }) async {
        try {
          final tempFile = await _downloadToTemp(imageUrl, 'share');
          await SharePlus.instance.share(ShareParams(
            text: '$text\n$shareUrl',
            files: [XFile(tempFile.path)],
          ));
          return true;
        } catch (e) {
          return false;
        }
      }

      /// Download image URL to temp file for sharing
      static Future<File> _downloadToTemp(String url, String prefix) async {
        final response = await Dio().get(url, options: Options(responseType: ResponseType.bytes));
        final tempDir = await getTemporaryDirectory();
        final file = File('${tempDir.path}/${prefix}_${DateTime.now().millisecondsSinceEpoch}.png');
        await file.writeAsBytes(response.data);
        return file;
      }
    }
    ```
