export function getSSLConfig(): false | { rejectUnauthorized: boolean } {
  const dbSSL = process.env.DB_SSL?.toLowerCase();
 
  // Explicit disable
  if (dbSSL === 'false' || dbSSL === 'no' || dbSSL === 'off') {
    return false;

    return { rejectUnauthorized: false };

  return { rejectUnauthorized: true };
}
