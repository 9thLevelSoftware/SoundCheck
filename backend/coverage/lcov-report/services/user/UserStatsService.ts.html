          uniqueVenues: 0,
          uniqueBands: 0,
        };
      }

      return {
        totalCheckins: parseInt(stats.checkin_count, 10) || 0,
        badgesEarned: parseInt(stats.badge_count, 10) || 0,
        followersCount: parseInt(stats.follower_count, 10) || 0,

    }
  }
